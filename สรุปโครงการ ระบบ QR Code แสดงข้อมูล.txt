    var id = e.parameter.id;
    if (!id) return createJsonResponse({ status: "error", message: "Missing product ID" });
    
    var sheet = SpreadsheetApp.getActiveSpreadsheet().getActiveSheet();
    var data = sheet.getDataRange().getValues();
    var headers = data[0];
    var productData = null;
    
    for (var i = 1; i < data.length; i++) {
      if (data[i][0].toString().trim() === id.toString().trim()) {
        productData = {};
        for (var j = 0; j < headers.length; j++) {
          productData[headers[j]] = data[i][j];
        }
        break;
      }
    }
    
    if (productData) {
      if (productData.date instanceof Date) {
        productData.date = Utilities.formatDate(productData.date, Session.getScriptTimeZone(), "yyyy-MM-dd");
      }
      return createJsonResponse({ status: "success", data: productData });
    } else {
      return createJsonResponse({ status: "error", message: "Product not found" });
    }
  } catch (err) {
    return createJsonResponse({ status: "error", message: err.toString() });
  }
}

function createJsonResponse(object) {
